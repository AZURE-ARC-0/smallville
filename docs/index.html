Getting Started

Below is an example of how to create custom generative agents

                    
    import { Smallville } from 'smallville'
                        
    const smallville = new Smallville({
        host: 'http://localhost:8080',
        stateHandler: (state) => {
            console.log(state)
        }
    });
                        
    const successful = await smallville.init()
                        
    if (successful) {
        startSimulation()
    }
                        
    async function startSimulation() {
        await smallville.createLocation({
            name: 'Forest',
        })

        await smallville.createObject({
            parent: 'Forest',
            name: 'Campfire',
            state: 'on'
        })
                            
        await smallville.createObject({
            parent: 'Forest',
            name: 'Branches',
            state: 'broken'
        })

        await smallville.createAgent({
            name: "Medina",
            location: "Forest: Campfire",
            activity: "Staying warm by the campfire",
            memories: [
                    "Initial Characteristics go here"
                ]
        })

        await smallville.updateState()                        
    }
